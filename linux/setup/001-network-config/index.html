    001-network-config

    ubuntu server wifi¶

    Connect to Wi-Fi From Terminal on Ubuntu 22.04/20.04 with WPA Supplicant

    sudo systemctl stop NetworkManager
systemctl list-unit-files  --type=service | grep enabled

    netplan dhcp¶

    echo "
network:
  ethernets:
    eth0:
      dhcp4: true
  version: 2
" | sudo tee /etc/netplan/00-installer-config.yaml

    netplan static ip¶

    echo "
network:
  ethernets:
    eth0:
      addresses:
        - 192.168.1.11/24
      gateway4: 192.168.1.1
      nameservers:
        addresses: [192.168.1.1]
  version: 2
" | sudo tee /etc/netplan/00-installer-config.yaml

    netplan wifi¶

    network:
  version: 2
  wifis:
    wlp2s0:
      access-points:
        wifi-ssid:
          password: passw0rd
      dhcp4: true

    netplan wifi¶

    network:
  version: 2
  wifis:
    wlp2s0:
      access-points:
        wifi-ssid:
          password: passw0rd
      dhcp4: false
      addresses:
        - 192.168.1.12/24
      routes:
        - to: default
          via: 192.168.1.1
      nameservers:
        addresses: [192.168.1.1]
